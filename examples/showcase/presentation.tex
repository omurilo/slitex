\begin{frame}{Theorem-like Environments}
  \begin{theorem}[Pythagorean Theorem]
    In a right triangle: $a^2 + b^2 = c^2$.
  \end{theorem}

  \begin{definition}
    A \textbf{prime} is a natural number $p > 1$ with no divisors other than $1$ and $p$.
  \end{definition}

  \begin{lemma}
    Every integer $n > 1$ has a prime factor.
  \end{lemma}

  \begin{proof}
    By strong induction on $n$. The details are left as an exercise.
  \end{proof}
\end{frame}

\begin{frame}{More Theorem Environments}
  \begin{corollary}
    There are infinitely many prime numbers.
  \end{corollary}

  \begin{remark}

  \end{remark}

  \begin{example}
    $2, 3, 5, 7, 11, 13, \ldots$ are all prime.
  \end{example}

  \begin{proposition}
    Every even integer greater than 2 is the sum of two primes (Goldbach's conjecture).
  \end{proposition}
\end{frame}

    \nabla \cdot \mathbf{E}  &= \frac{\rho}{\varepsilon_0}     \\

    \nabla \times \mathbf{E} &= -\frac{\partial \mathbf{B}}{\partial t} \\

\begin{frame}{Column with Math \& Text}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Gradient Descent}

      Update rule:
      \[
        \theta_{t+1} = \theta_t - \alpha \nabla_\theta \mathcal{L}(\theta_t)
      \]
      where $\alpha$ is the learning rate.
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{block}{Parameters}
        \begin{itemize}
          \item $\alpha$: learning rate
          \item $\theta$: model weights
          \item $\mathcal{L}$: loss function
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Overlay — \texttt{\textbackslash only}}
  \only<1>{
    \textbf{Step 1:} Define the problem.

    Clearly state what you want to achieve.
  }
  \only<2>{
    \textbf{Step 2:} Design the solution.

    Break the problem into manageable parts.
  }
  \only<3>{
    \textbf{Step 3:} Implement and test.

    Write the code; verify against your requirements.
  }

\begin{frame}{Overlay — \texttt{\textbackslash uncover}}
  \uncover<1->{Always visible from step 1.}

  \uncover<2->{Revealed at step 2, stays visible.}

  \uncover<3->{Revealed at step 3, stays visible.}

  \uncover<4->{Revealed at step 4, stays visible.}
\end{frame>

\begin{frame}{Overlay --- List Items}
  \begin{itemize}

    \item<3-> Third item (from step 3)
    \item<4-> Fourth item (from step 4)

\end{frame}

\begin{frame}{Overlay --- Specific Steps}

    \item<2> Only on step 2
    \item<1,3> On steps 1 and 3
    \item<2-4> On steps 2 through 4
