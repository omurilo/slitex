import numpy as np

def softmax(x: np.ndarray) -> np.ndarray:
    e = np.exp(x - x.max())
    return e / e.sum()

probs = softmax(np.array([1.0, 2.0, 3.0]))
print(probs)  # [0.090, 0.245, 0.665]

def gd(grad, x0, lr=0.01, steps=100):
    x = x0
    for _ in range(steps):
        x = x - lr * grad(x)
    return x
